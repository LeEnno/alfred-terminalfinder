on alfred_script(q)	tell application "Finder"		set pathList to (quoted form of POSIX path of (folder of the front window as alias))		set command to "clear; cd " & pathList	end tell	tell application "iTerm"		if it is running then			set app_running to true		else			set app_running to false		end if		activate		set hasNoWindows to ((count of windows) = 0)		if app_running and hasNoWindows then			create window with default profile		end if		repeat			if ((count of windows) > 0) then				exit repeat			end if		end repeat		select first window		tell the first window			if app_running and hasNoWindows is false then				create tab with default profile			end if			tell current session to write text command		end tell	end tellend alfred_script
